#views.py
from .forms import BookForm
def order(request):
    form = BookForm()
    return render(request,'myapp/order.html',{'form':form})

#forms.py
from django import forms
from .models import Book

class BookForm(forms.ModelForm):
    class Meta:
        model=Book
        fields=['title']
	#fields='__all__'
        labels={'title':'Title'}
        widgets={'title':forms.Textarea}
